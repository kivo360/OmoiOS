    <script>
        const API_BASE = window.location.origin;
        let ws = null;
        let currentTicketId = null;

        // Initialize
        document.addEventListener('DOMContentLoaded', () => {
            loadTickets();
            connectWebSocket();
        });

        // WebSocket connection
        function connectWebSocket() {
            const wsUrl = `${API_BASE.replace('http', 'ws')}/api/v1/ws/events`;
            ws = new WebSocket(wsUrl);

            ws.onopen = () => {
                document.getElementById('wsStatus').classList.remove('disconnected');
                document.getElementById('wsStatusText').textContent = 'Connected';
            };

            ws.onclose = () => {
                document.getElementById('wsStatus').classList.add('disconnected');
                document.getElementById('wsStatusText').textContent = 'Disconnected';
                // Reconnect after 3 seconds
                setTimeout(connectWebSocket, 3000);
            };

            ws.onmessage = (event) => {
                const data = JSON.parse(event.data);
                handleEvent(data);
            };
        }

        function handleEvent(event) {
            if (event.type === 'ping') return;

            const output = document.getElementById('outputArea');
            const line = document.createElement('div');
            line.className = 'output-line system';
            line.textContent = `[${event.event_type}] ${event.entity_type}: ${event.entity_id?.slice(0, 8)}...`;
            output.appendChild(line);
            output.scrollTop = output.scrollHeight;

            // Refresh tickets on relevant events
            if (['TASK_COMPLETED', 'TASK_FAILED', 'TICKET_RESOLVED'].includes(event.event_type)) {
                loadTickets();
            }
        }

        // Load tickets
        async function loadTickets() {
            try {
                const response = await fetch(`${API_BASE}/api/v1/tickets/?limit=10`);
                const data = await response.json();
                renderTickets(data.tickets || []);
            } catch (error) {
                console.error('Failed to load tickets:', error);
                // Try alternative endpoint
                try {
                    const response = await fetch(`${API_BASE}/api/v1/tickets/list`);
                    const data = await response.json();
                    renderTickets(data || []);
                } catch (e) {
                    document.getElementById('ticketList').innerHTML = `
                        <div class="empty-state">Failed to load tickets</div>
                    `;
                }
            }
        }

        function renderTickets(tickets) {
            const container = document.getElementById('ticketList');

            if (!tickets.length) {
                container.innerHTML = `
                    <div class="empty-state">No tickets yet</div>
                `;
                return;
            }

            container.innerHTML = tickets.map(ticket => `
                <div class="ticket-item ${ticket.id === currentTicketId ? 'active' : ''}"
                     onclick="selectTicket('${ticket.id}')">
                    <div class="ticket-title">${escapeHtml(ticket.title)}</div>
                    <div class="ticket-meta">
                        <span class="status-tag status-${getStatusClass(ticket.status)}">${ticket.status}</span>
                        <span>${formatDate(ticket.created_at)}</span>
                    </div>
                </div>
            `).join('');
        }

        function selectTicket(ticketId) {
            currentTicketId = ticketId;
            loadTickets(); // Refresh to show active state
            loadTicketDetails(ticketId);
        }

        async function loadTicketDetails(ticketId) {
            try {
                const response = await fetch(`${API_BASE}/api/v1/tickets/${ticketId}`);
                const ticket = await response.json();

                document.getElementById('outputTitle').textContent = `Ticket: ${ticket.title?.slice(0, 40)}...`;

                const output = document.getElementById('outputArea');
                output.innerHTML = '';

                addOutputLine(`Ticket ID: ${ticket.id}`, 'info');
                addOutputLine(`Status: ${ticket.status}`, 'info');
                addOutputLine(`Priority: ${ticket.priority}`, 'info');
                addOutputLine(`Created: ${formatDate(ticket.created_at)}`, 'info');
                addOutputLine('', 'info');
                addOutputLine(`Description:`, 'info');
                addOutputLine(ticket.description || 'No description', 'info');

            } catch (error) {
                addOutputLine(`Failed to load ticket: ${error.message}`, 'error');
            }
        }

        // Submit task
        async function submitTask() {
            const prompt = document.getElementById('promptInput').value.trim();
            if (!prompt) {
                alert('Please enter a task description');
                return;
            }

            const runBtn = document.getElementById('runBtn');
            runBtn.disabled = true;
            runBtn.textContent = 'Running...';

            const output = document.getElementById('outputArea');
            output.innerHTML = '';
            document.getElementById('outputTitle').textContent = 'Running Task...';

            addOutputLine('ðŸš€ Creating ticket...', 'system');

            try {
                // Create ticket
                const createResponse = await fetch(`${API_BASE}/api/v1/tickets/`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        title: prompt.slice(0, 100),
                        description: prompt,
                        priority: 'MEDIUM',
                        phase_id: 'PHASE_IMPLEMENTATION'
                    })
                });

                if (!createResponse.ok) {
                    throw new Error(`Failed to create ticket: ${createResponse.status}`);
                }

                const ticket = await createResponse.json();
                currentTicketId = ticket.id;
                addOutputLine(`âœ… Ticket created: ${ticket.id}`, 'success');

                // Check if ticket needs approval
                addOutputLine('ðŸ“‹ Checking ticket status...', 'system');
                const ticketDetails = await fetch(`${API_BASE}/api/v1/tickets/${ticket.id}`);
                const ticketData = await ticketDetails.json();

                if (ticketData.approval_status === 'pending' || ticketData.approval_status === 'pending_review') {
                    // Approve ticket
                    addOutputLine('ðŸ“‹ Approving ticket...', 'system');
                    const approveResponse = await fetch(`${API_BASE}/api/v1/tickets/approve`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ ticket_id: ticket.id })
                    });

                    if (!approveResponse.ok) {
                        const errorData = await approveResponse.json();
                        addOutputLine(`âš ï¸ Approval: ${errorData.detail || 'Already approved'}`, 'info');
                    } else {
                        addOutputLine('âœ… Ticket approved, task created', 'success');
                    }
                } else {
                    addOutputLine('âœ… Ticket auto-approved', 'success');
                }
                addOutputLine('', 'info');
                addOutputLine('â³ Waiting for agent to pick up task...', 'system');
                addOutputLine('(The orchestrator assigns tasks to available workers)', 'info');

                // Refresh tickets
                loadTickets();

                document.getElementById('outputTitle').textContent = `Task: ${prompt.slice(0, 40)}...`;

            } catch (error) {
                addOutputLine(`âŒ Error: ${error.message}`, 'error');
            } finally {
                runBtn.disabled = false;
                runBtn.textContent = 'Run Task';
            }
        }

        // Utility functions
        function addOutputLine(text, className = '') {
            const output = document.getElementById('outputArea');
            const line = document.createElement('div');
            line.className = `output-line ${className}`;
            line.textContent = text;
            output.appendChild(line);
            output.scrollTop = output.scrollHeight;
        }

        function escapeHtml(text) {
            const div = document.createElement('div');
            div.textContent = text || '';
            return div.innerHTML;
        }

        function getStatusClass(status) {
            const map = {
                'pending': 'pending',
                'backlog': 'pending',
                'analyzing': 'running',
                'building': 'running',
                'running': 'running',
                'completed': 'completed',
                'done': 'completed',
                'failed': 'failed'
            };
            return map[status] || 'pending';
        }

        function formatDate(dateStr) {
            if (!dateStr) return '';
            const date = new Date(dateStr);
            return date.toLocaleString();
        }

        // Handle Enter key in prompt
        document.getElementById('promptInput').addEventListener('keydown', (e) => {
            if (e.key === 'Enter' && e.metaKey) {
                submitTask();
            }
        });
    </script>
